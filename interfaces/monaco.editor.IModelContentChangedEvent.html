Interface IModelContentChangedEvent

Source
Expand description

An event describing a change in the text of a model.

interface IModelContentChangedEvent {
    changes: IModelContentChange[];
    eol: string;
    isEolChange: boolean;
    isFlush: boolean;
    isRedoing: boolean;
    isUndoing: boolean;
    versionId: number;
}
